命令行

登录sql：登录远程主机名/当前主机127.0.0.1可以省略-h
mysql -u root -p 登录MySQL数据库，输入密码
create database quanjing; 新增数据库quanjing【增】
drop database quanjing; 删除数据库quanjing【删】
show databases; 查看所有数据库，以分号结束命令【查】
use quanjing; change 到qunajing数据库
show tables; 查看quanjng数据库下所有表名（如skills）
create table skills (id int unsigned not null auto_increment primary key, skill_name char(8) not null, level tinyint unsigned not null default 2); 创建表skills，头信息和修饰语
select * from skills；查看skills表下所有信息【查】
select level, skill_name form skills; 查看skills表下level和skill_name列的信息
insert into skills (id, skill_name, level) values (), (), (); 向表skills插入行信息【增】
delete from skills where id=1;删除表信息【删】
drop table skills;删除表skills【删】
quit\exit退出sql
